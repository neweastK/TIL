import sys
input = sys.stdin.readline

N,M = map(int,input().split())

graph = [[] for _ in range(N+1)]

for _ in range(M):
    a,b = map(int,input().split())
    # aëŠ” bë¥¼ ì‹ ë¢°í•œë‹¤ == bì—ì„œ aë¡œ ì´ë™í•  ìˆ˜ ìˆë‹¤
    graph[b].append(a)


def dfs(start):
    stack = [start]
    visited = [0]*(N+1)
    visited[start] = 1
    cnt = 0

    while stack:
        now = stack.pop()

        for i in graph[now]:
            if visited[i] != 1:
                visited[i] = 1
                cnt+=1
                stack.append(i)
    return cnt

res = []
value = 0
for i in range(1,N+1):
    result = dfs(i)
    if result>value:
        res = [i]
        value = result
    elif result == value:
        res.append(i)
print(*res)
