def check(size):
    cnt=0
    tmp = 0
    for num in nums:

            tmp=num
            cnt+=1
        else:
            tmp+=num

    mid = (start+end)//2
    res = check(mid)
    if res>M:
        start=mid+1
    else:
        end=mid-1
        if answer>mid:
            answer = mid
