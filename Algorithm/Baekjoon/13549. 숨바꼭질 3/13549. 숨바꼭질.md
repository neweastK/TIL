import heapq

def dijkstra(start):
    queue = [(start,0)]
    distance[start] = 0
    
    while queue:
        target, dist = heapq.heappop(queue)

        if dist > distance[target]:
            continue
        
        for i in graph[target]:
            cost = i[1]+dist
            if cost<distance[i[0]]:
                distance[i[0]] = cost
                heapq.heappush(queue,(i[0],cost))

N,K = map(int,input().split())

    tmp = [(i-1,1),(i+1,1),(2*i,0)]
    graph[i] = tmp
